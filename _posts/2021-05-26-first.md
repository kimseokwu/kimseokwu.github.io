    for x_i, y_i in zip(x, y):
        y_hat = x_i*w+ b
        err = y_i - y_hat
        w_rate = x_i

# 선형 회귀를 위한 뉴런을 만듭니다.


```python
class Neuron:
    def __init__(self):
        self.w = 1.0
        self.b = 1.0
    
    def forpass(self, x):
        # 정방향 계산
        y_hat = x * self.w + self.b
        return y_hat
        
    def backprop(self, x, err):
        # 역방향 계산
        w_grad = x * err
        b_grad = 1 * err
        return w_grad, b_grad
    
    def fit(self, x, y, epochs=100):
        for i in range(epochs):
            for x_i, y_i in zip(x, y):
                y_hat = self.forpass(x_i)
                err = -(y_i - y_hat)
                w_grad, b_grad = self.backprop(x_i, err)
                self.w -= w_grad
                self.b -= b_grad
        
```


```python
neuron = Neuron()
neuron.fit(x, y)
```


```python
plt.scatter(x, y)
pt1 = (-0.1, -0.1* neuron.w + neuron.b )
pt2 = (0.15, 0.15*neuron.w + neuron.b)
plt.plot([pt1[0], pt2[0]], [pt1[1], pt2[1]])
plt.xlabel('x')
plt.ylabel('y')
plt.show()
```
